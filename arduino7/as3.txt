const int SW = 2;
const int X = 0;
const int Y = 1;
void setup() {
 pinMode(SW,INPUT_PULLUP);
Serial.begin(9600);
}

void loop() {
 Serial.print("Switch: ");
 Serial.print(digitalRead(SW));
Serial.print("\t\t");
Serial.print("VRX ");
Serial.print(analogRead(SW));
Serial.print("\t\t");
Serial.print("VRX ");
Serial.print(analogRead(Y));
Serial.println("___________");
delay(500);
}
